--    1. Показать всю информацию о продажах.
use Musicshop; 
SELECT CD.name AS [Имя дисков],CD.CD_Date AS [Год выпуска], Format.name AS [Формат], Sl.name AS [Имя продавца]
FROM dbo.CD
INNER JOIN dbo.Format ON Format.id=CD.id
INNER JOIN dbo.Selling AS S ON S.id_cd=CD.id
INNER JOIN dbo.Sellers AS Sl ON Sl.id=S.id_seller
GO

--   2. Показать кол-во проданных дисков по каждой из групп. 
use Musicshop;
SELECT COUNT (S.id_cd) AS [Количество проданных], B.name AS [Aльбом]
FROM dbo.Selling AS S
INNER JOIN dbo.CD  ON CD.id=S.id_cd
INNER JOIN dbo.Bands AS B ON B.id=CD.id_band
GROUP BY B.name
ORDER BY COUNT(S.id_cd)
